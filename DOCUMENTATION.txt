/*
	Function: _grid = ["new", [xmapsize,ymapsize,xsectorsize,ysectorsize]] call OO_GRID;
	Create a new grid object
	
	Parameters:
		xmapsize - X map size
		ymapsize - Y map size
		xsectorsize - X sector size
		ysectorsize - X sector size
	
*/

/*
	Function: = ["delete", _grid] call OO_GRID;
	Delete the grid object
	
	Parameters:
		grid - OO GRID object.
	
*/

/*
	Function: ["parseAllSectors", string] spawn _grid;
	Parse all sectors of grid with a function call that returns true or false
	
	Parameters:
		string - function name
	
*/


/*
	Function: ["getSectorFromPos", array] spawn _grid;
	Get sector from a position
	
	Parameters:
		array - position
	
*/

/*
	Function: ["getSectorAround", array] spawn _grid;
	Get an array of sectors all around the sector (3x3 sectors)
	
	Parameters:
		array - sector position
	
*/

/*
	Function: ["getCenterPos", array] spawn _grid;
	Get center position of a sector with a position
	
	Parameters:
		array - position

*/

/*
	Function: ["getSectorAllAround", array] spawn _grid;
	Get an array of sectors all around the sector at x scales (scale x scale sectors)
	
	Parameters:
		array 
			- sector position
			- scale

	
*/

/*
	Function: ["getSectorCrossAround", array] spawn _grid;
	Get an array of cross sectors all around the sector 2 horizontals + 2 verticals + 1 center
	
	Parameters:
		array - sector position

*/

/*
	Function: ["getPosFromSector", array] spawn _grid;
	Get a position from a sector
	
	Parameters:
		array - sector position
	
*/
